SELECT I.ID, I.SUMMARY, I.DESCRIPTION, IL.NAME
FROM ISSUES I JOIN ISSUESLIST IL ON I.ISSUESLIST_ID = IL.ID;

SELECT I.ID, I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_ISSUES_ASSIGNED_TO
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING COUNT(*) > 0;
